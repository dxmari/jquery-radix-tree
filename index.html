    Guides: Usage & Demos
    Basic Example
    With Radix Tree, you can render simple tree deeply nestable nodes with a basic hierarchical data structure. You can expand and collapse nodes, and the tree supports unlimited nesting. Useful for representing folders, categories, or any parent-child relationships.
    For example, you can use import this snippet inside your files like this:
    Basic Demo
                
const data = [
  {
    label: 'Universe',
    open: true,
    children: [
      { label: 'Galaxies', children: [ { label: 'Milky Way' }, { label: 'Andromeda' } ] },
      { label: 'Black Holes' }
    ]
  }
];
$('.radix-tree').radixTree({ data });
    Output
    Lazy Loading
    Load children on demand with async support. This example shows how to use the lazyLoad callback to fetch children only when a node is expanded. Ideal for large datasets or when loading data from a server.
    Lazy Loading Demo
                
function myLazyLoad(node, done, delay) {
    setTimeout(() => {
      if (node.label === 'Galaxies') {
        done([{ label: 'Milky Way' }, { label: 'Andromeda' }]);
      } else {
        done([{ label: 'No data' }]);
      }
    }, delay); // Use the dynamic delay
  }
$('.radix-tree').radixTree({
  data: [{ label: 'Galaxies', lazy: true }],
  lazyLoad: myLazyLoad,
  lazyLoadDelay: 1200 // Set delay dynamically (ms)
});

    Infinite Scroll
    Efficiently browse huge folders or datasets. This example shows how to enable infinite scroll and pagination for large trees, loading more nodes as you scroll. Useful for file browsers or any large hierarchical data.
    Infinite Scroll Demo
                
const demoData = [
    {
      label: 'Big Folder',
      open: true,
      infinite: true,
      lazy: true,
      badge: 100,
      tags: ['infinite', 'files']
    }
  ];
function infiniteLazyLoad(node, done, opts, delay) {
  const total = 100;
  const page = opts && opts.page ? opts.page : 1;
  const pageSize = opts && opts.pageSize ? opts.pageSize : 20;
  const start = (page - 1) * pageSize;
  const end = Math.min(start + pageSize, total);
  const children = [];
  for (let i = start; i < end; i++) {
    children.push({
      label: 'File ' + (i + 1),
      badge: (i % 2 === 0) ? 'even' : 'odd',
      tags: (i % 10 === 0) ? ['milestone'] : []
    });
  }
  setTimeout(() => {
    done(children, end < total);
  }, delay); // Use the dynamic delay
}
// Example 1: Small pageSize, default threshold (pagination enabled for 5)
$('.radix-tree-small').radixTree({
  data: demoData,
  lazyLoad: infiniteLazyLoad,
  pageSize: 5,         // Pagination enabled for 5 (default threshold is 5)
  lazyLoadDelay: 800
});
// Example 2: Custom threshold (pagination only for pageSize >= 10)
$('.radix-tree-large').radixTree({
  data: demoData,
  lazyLoad: infiniteLazyLoad,
  pageSize: 5,         // Pagination NOT enabled for 5 (threshold is 10)
  paginateThreshold: 10,
  lazyLoadDelay: 800
});

    Command API
    Interact with the tree after initialization. This example covers the available API commands for getting and setting checked nodes, expanding/collapsing nodes, and updating the tree data programmatically.
    Command API Demo
                
// Get all checked nodes
const checked = $('.radix-tree').radixTree('getChecked');
// Set a node as checked
$('.radix-tree').radixTree('setChecked', nodeId, true);
// Expand/collapse nodes programmatically
$('.radix-tree').radixTree('expand', nodeId);
$('.radix-tree').radixTree('collapse', nodeId);
// Get or set the entire data
const currentData = $('.radix-tree').radixTree('getData');
